DEFINE FUNCTION main_menu
	WHILE True 
		PRINT (*"Welcome to the University Management System")
		PRINT ("1. Administrator")
		PRINT ("2. Lecturer")
		PRINT ("3. Student Management")
		PRINT ("4. Registrator")
		PRINT ("5. Accountant")
		PRINT ("6. Exit")
		PRINT ("What is your role?")
		GET choice
		IF choice EQUALS 1 THEN
			CALL FUNCTION administrator_login()
		ELIF choice EQUALS 2 THEN
			CALL FUNCTION lecturer_login()
		ELIF choice EQUALS 3 THEN
			CALL FUNCTION student_management_login()
		ELIF choice EQUALS 4 THEN 
			CALL FUNCTION registrator_login()
		ELIF choice EQUALS  5 THEN
			CALL FUNCTION accountant_login()
		ELIF choice EQUALS 6 THEN
			PRINT("Exiting the system. Goodbye!")
			BREAK
		ELSE
			PRINT("Invalid choice. Please try again.")
		END IF
	END WHILE
END DEFINE
			

DEFINE FUNCTION login(role_code, calling_menu)
	SET attempts TO 3
	WHILE attempts > 0
		PRINT("You need to enter the access code to your role")
		GET access_code_input 
		IF access_code_input IS NUMERIC THEN
			IF access_code_input EQUALS role_code THEN
				CALL FUNCTION calling_menu()
				RETURN
			ELSE
				attempts EQUALS attempts - 1
				PRINT("Incorrect code. You have " +  CONVERT attempts TO STRING +" attempt(s) left.")
			END IF 
		ELSE
			PRINT("Invalid input. Please enter a number.")
		END IF
	END WHILE 
	PRINT("Access denied, kicking you from the system.")
	EXIT
END DEFINE

DEFINE FUNCTION count_lines_with_values(file_path)
	OPEN file file_path IN READ MODE AS file 
	SET line_count TO 0
	FOR EACH line IN file 
		IF line.strip() IS NOT EMPTY THEN
			SET line_count TO line_count + 1
		END IF 
	END FOR
	RETURN line_count
END DEFINE 

DEFINE FUNCTION cancel_command(input_value)
    IF input_value.strip().upper() EQUALS "CANCEL" THEN
        RETURN True
    ELSE
        RETURN False
    END IF
END DEFINE

DEFINE FUNCTION null_input(input_value)
    IF input_value EQUALS "" THEN
        PRINT("Input cannot be null, Please enter a valid input!")
        RETURN True
    ELSE
        RETURN False
    END IF
END DEFINE

DEFINE FUNCTION add_course_code(courses)
    WHILE True
        PRINT("Please enter the new course code :")
        GET new_course_code
        IF null_input(new_course_code) THEN
            CONTINUE
        ELSE IF cancel_command(new_course_code) THEN
            RETURN None
        END IF
        FOR EACH course IN courses
            IF course[0] EQUALS new_course_code THEN
                PRINT("Invalid input! (" + new_course_code + " already exists).")
                BREAK
            END IF
        END FOR
        IF file_exists("courses.txt") THEN
            OPEN file "courses.txt" IN READ MODE AS course_files
            SET found TO False
            FOR EACH course IN course_files
                SPLIT course BY "," INTO data
                SET course_code_from_file TO data[0].strip()
                IF course_code_from_file EQUALS new_course_code THEN
                    PRINT("Invalid input! (" + new_course_code + " already exists).")
                    SET found TO True
                    BREAK
                END IF
            END FOR
            IF NOT found THEN
                RETURN new_course_code
            END IF
        ELSE
            RETURN new_course_code
        END IF
    END WHILE
END DEFINE


DEFINE FUNCTION add_course_name
    WHILE True
        PRINT("Please enter the name of the new course: ")
        GET new_course_name
        IF null_input(new_course_name) THEN
            CONTINUE
        ELSE IF cancel_command(new_course_name) THEN
            RETURN None
        ELSE IF any(char IS NUMERIC FOR char IN new_course_name) THEN
            PRINT("Course name must be alphabetic. Please try again!")
            CONTINUE
        ELSE
            RETURN new_course_name
        END IF
    END WHILE
END DEFINE

DEFINE FUNCTION add_course_credits
    WHILE True
        PRINT("Please enter the new course credits: ")
        GET new_course_credits
        IF null_input(new_course_credits) THEN
            CONTINUE
        ELSE IF cancel_command(new_course_credits) THEN
            RETURN None
        ELSE IF new_course_credits.isnumeric() THEN
            RETURN new_course_credits
        ELSE
            PRINT("Credits must be numeric. Please try again!")
        END IF
    END WHILE
END DEFINE

DEFINE FUNCTION add_course(courses)
    PRINT("(If you want to cancel, type 'cancel' at any step to stop adding the course.)")
    SET new_course_code TO CALL FUNCTION add_course_code(courses)
    IF new_course_code IS None THEN
        RETURN
    END IF
    SET new_course_name TO CALL FUNCTION add_course_name()
    IF new_course_name IS None THEN
        RETURN
    END IF
    SET new_course_credits TO CALL FUNCTION add_course_credits()
    IF new_course_credits IS None THEN
        RETURN
    END IF
    APPEND [new_course_code, new_course_name, new_course_credits] TO courses
    PRINT("Course " + new_course_name + " added successfully.")
    OPEN file "courses.txt" IN APPEND MODE AS file
    WRITE new_course_code + "," + new_course_name + "," + new_course_credits + "\n" TO file
END DEFINE

DEFINE FUNCTION load_courses(courses_list)
    IF file_exists("courses.txt") THEN
        OPEN file "courses.txt" IN READ MODE AS file
        FOR EACH line IN file
            SPLIT line BY "," INTO course_code, course_name, course_credits
            APPEND [course_code, course_name, course_credits] TO courses_list
        END FOR
        RETURN True
    ELSE
        PRINT("Courses file is not found in the system.")
        RETURN False
    END IF
END DEFINE

DEFINE FUNCTION display_courses(courses_list)
    IF NOT courses_list THEN
        PRINT("\nNo courses available")
        RETURN False
    END IF
    PRINT("Below is the list of the courses:\n")
    FOR EACH index, course IN ENUMERATE(courses_list)
        PRINT(index + 1 + ". Course Code: " + course[0] + "\n   Course Name: " + course[1] + "\n   Credits: " + course[2])
    END FOR
    RETURN True
END DEFINE

DEFINE FUNCTION save_courses(courses_list)
    OPEN file "courses.txt" IN WRITE MODE AS file
    FOR EACH course IN courses_list
        WRITE JOIN(course, ",") + "\n" TO file
    END FOR
END DEFINE

DEFINE FUNCTION remove_courses(courses_list)
    SET file_loaded TO CALL FUNCTION load_courses(courses_list)
    IF NOT file_loaded THEN
        RETURN
    END IF
    IF NOT CALL FUNCTION display_courses(courses_list) THEN
        RETURN
    END IF
    WHILE True
        PRINT("Enter the number of the course you want to remove: ")
        GET courses_input
        SET courses_input TO courses_input - 1
        IF IS_NUMERIC(courses_input) THEN
            IF courses_input < 0 OR courses_input >= LENGTH(courses_list) THEN
                PRINT("Invalid input, please try again!")
                CONTINUE
            END IF
            SET selected_course TO courses_list[courses_input]
            PRINT("Selected Course: " + selected_course[0])
            WHILE True
                PRINT("Are you sure you want to remove " + selected_course[0] + "? (Y/N): ")
                GET confirm
                IF confirm EQUALS 'Y' THEN
                    REMOVE courses_list[courses_input]
                    PRINT("Course " + selected_course[0] + " removed successfully.")
                    CALL FUNCTION save_courses(courses_list)
                    RETURN
                ELSE IF confirm EQUALS 'N' THEN
                    PRINT("Removal canceled.")
                    RETURN
                ELSE
                    PRINT("Invalid input. Please enter 'Y' or 'N'.")
                END IF
            END WHILE
        ELSE
            PRINT("Invalid input. Please enter a valid number.")
        END IF
    END WHILE
END DEFINE

DEFINE FUNCTION add_lectures_course_code
    WHILE True
        PRINT("\nPlease make sure to separate multiple course codes with '|'")
        PRINT("For example: CS001|IT002|IN003")
        PRINT("Enter 'cancel' if you want to cancel the process.\n")
        PRINT("Please enter the lecturer's course code: ")
        GET lecturers_course_code
        IF null_input(lecturers_course_code) THEN
            CONTINUE
        ELSE IF cancel_command(lecturers_course_code) THEN
            RETURN None
        END IF

		DECLARE validated_course_codes as empty list
        IF file_exists("courses.txt") THEN
            OPEN file "courses.txt" IN READ MODE AS courses_file
            FOR EACH line IN courses_file
                SPLIT line BY "," INTO data
                IF data IS NOT EMPTY THEN
                    APPEND data[0].upper() TO validated_course_codes
                END IF
            END FOR

            SET entered_course_codes TO SPLIT lecturers_course_code BY "|"
            SET invalid_course_codes TO [code FOR code IN entered_course_codes IF code NOT IN validated_course_codes]
            IF invalid_course_codes IS NOT EMPTY THEN
                PRINT("The following course codes are invalid or do not exist: " + JOIN(invalid_course_codes, ", "))
                CONTINUE
            END IF

            DECLARE validated_lecturers_cc as empty list
            OPEN file "lecturers.txt" IN READ MODE AS lecturers_file
            FOR EACH line IN lecturers_file
                SPLIT line BY "," INTO data
                IF LENGTH(data) > 4 THEN
                    SET lecturer_course_codes TO SPLIT data[4].upper() BY "|"
                    FOR EACH code IN lecturer_course_codes
                        APPEND code TO validated_lecturers_cc
                    END FOR
                END IF
            END FOR
            SET invalid_lecturers_codes TO [code FOR code IN entered_course_codes IF code IN validated_lecturers_cc]
            IF invalid_lecturers_codes IS NOT EMPTY THEN
                PRINT("The following course codes are already assigned to other lecturers: " + JOIN(invalid_lecturers_codes, ", "))
                CONTINUE
            ELSE
                RETURN lecturers_course_code
            END IF
        ELSE
            PRINT("\nFile does not exist, creating one right now.\nPlease try using the functions now.")
            OPEN FILE "lecturers.txt" IN WRITE MODE AS create_file
        END IF
    END WHILE
END DEFINE

DEFINE FUNCTION record_grades(course_codes, student_info)

	WHILE True
		PRINT("Enter student's grade (input 'cancel' to cancel): ")
		GET new_grade 
		IF IS_DIGIT(new_grade) AND 0 <= CONVERT new_grade TO INTEGER <=100 THEN
			OPEN file ENROLLMENTS_FILE IN READ MODE AS file 
			DECLARE lines AS LIST 
			FOR EACH line IN file
				APPEND line TO lines
			END FOR 
			DECLARE updated_lines AS LIST 
			SET record_found TO False
			FOR EACH line IN lines  
				SPLIT line BY "," INTO data 
				IF data[0] EQUALS student_info[0] THEN

DEFINE FUNCTION group_students_by_modules(lecturer_courses)
	DECLARE grouped_students AS LIST 
	DECLARE students_lookup AS LIST 
	OPEN file ENROLLMENTS_FILE IN READ MODE AS file
	DECLARED lines AS LIST
	FOR EACH line IN file 
		APPEND line TO lines 
	END FOR 
	FOR EACH line In lines 
		EVAL line.strip() INTO data 
		SET student_id TO data[0]
		SET student_name TO data[1]
		APPEND (student_id, student_name) TO students_lookup 
		FOR EACH course_data IN data[2:] 
			SET course_code TO course_data[0]
			IF course_code IN lecturer_courses THEN 
				SET course_found TO False
				FOR i FROM 0 TO LENGTH OF grouped_students - 1
					IF grouped_students[i][0] EQUALS course_code THEN 
						APPEND student_id TO grouped_students[i][1]
						SET course_found TO True
						BREAK LOOP 
					END IF
				END FOR 
				IF NOT course_found THEN
					APPEND (course_code, [student_id]) TO grouped_students
				END IF 
			END IF
		END FOR
	END FOR 
	RETURN grouped_students, students_lookup
END DEFINE

DEFINE FUNCTION mark_attendance(course_codes, student_info)
	OPEN file ENROLLMENTS_FILE IN READ MODE AS file 
	FOR EACH line IN file 
		APPEND line TO lines
	END FOR 
	DECLARE updated_lines AS LIST
	SET student_found TO False 
	FOR EACH line IN lines
		EVAL line.strip() INTO data
		SET student_id TO data[0]
		SET student_name TO data[1]
		SET attendance_courses TO data[2:] 
		IF student_id EQUALS student_info[0] THEN 
			SET student_found TO True
			SET course_updated TO False 
			FOR EACH course IN attendance_courses
				IF course[0] EQUALS course_codes THEN
					SET present TO course[2]
					SET absent TO course[3]
					SET valid TO False
					SET choice TO ""
					WHILE NOT valid
						PRINT("Is " + student_name + " present in " + course_codes + " [Y/N]? ")
						GET choice
						IF choice EQUALS "Y" THEN
							SET present TO present + 1
							SET valid TO True
						ELSE IF choice EQUALS "N" THEN
							SET absent TO absent + 1 
							SET valid TO True
						ELSE 
							PRINT("Invalid input. Please enter 'Y' or 'N'.")
						END IF
					END WHILE 
					SET course[2] TO present
					SET course[3] TO absent 
					SET course_updated TO True
					PRINT("[" + ('ABSENT' IF choice EQUALS 'N' ELSE 'PRESENT') + "] " + student_name + " in Module " + course_codes + ".")
				END IF
			END FOR
			IF NOT course_updated THEN
				PRINT("No enrollment found for " + student_name + " in " + course_codes + ".")
			END IF 
		END IF
		APPEND str(data) + "\n" TO updated_lines
	END FOR
	IF NOT student_found THEN
		PRINT("Student " + student_info[1] + " (ID: " + student_info[0] + ") not found in enrollments")
	END IF 
	OPEN file ENROLLMENTS_FILE IN WRITE MODE AS file
	FOR EACH updated_line IN updated_lines 
		WRITE updated_line TO file
	END FOR
END DEFINE

DEFINE FUNCTION enroll_student
	PRINT("Enroll Student to a Course")
	WHILE True
		PRINT("Student ID: ")
		GET enroll_student_id
		PRINT("Student Name: ")
		GET enroll_student_name
		DEFINE validate_students AS LIST
		OPEN file STUDENTS_FILE IN READ MODE AS student_file
		FOR EACH line IN student_file
			SPLIT line BY "," INTO data 
			APPEND data TO validate_students
		END FOR
		FOR EACH data IN validate_students
			IF data[0] EQUALS enroll_student_id AND data[1] EQUAL enroll_student_name THEN
				OPEN file ENROLLMENTS_FILE IN READ MODE AS enrollment_file
				FOR EACH line IN enrollment_file
					IF line IS NOT EMPTY THEN
						SPLIT line BY "," INTO data 
						IF data[0] EQUALS enroll_student_id AND data[1] EQUAL enroll_student_name THEN
							PRINT("There is already data for this student in the system")
							CALL FUNCTION continue_back_exit
							RETURN
						END IF
					END IF
				END FOR 
				DEFINE validate_courses AS LIST
				OPEN file COURSE_FILE IN READ MODE AS course_file
				FOR EACH line IN course_file
					SPLIT line BY "," INTO data
					APPEND data[0] TO validate_courses
				END FOR 
				DEFINE enrollment_record AS LIST 
				APPEND enroll_student_id TO enrollment_record 
				APPEND enroll_student_name TO enrollment_record
				SET total_student_course TO 5
				FOR i FROM 0 TO total_student_course - 1
					WHILE True 
						PRINT("Enter course code for course" + (i + i) + ": ")
						GET enroll_course_code
						IF enroll_course_code NOT IN validate_courses THEN
							PRINT("Error: Invalid course code.")
						ELSE 
							APPEND [enroll_course_code, None, 0,0] TO enrollment_record
							BREAK LOOP
						END IF
					END WHILE 
				END FOR 
				OPEN file ENROLLMENTS_FILE IN APPEND MODE AS enrollments_file
				WRITE enrollment_record JOINED BY "," TO enrollments_file
				PRINT("Successfully enrolled student in course")
				CALL FUNCTION continue_back_exit
				RETURN
			END IF
		END FOR 
		PRINT("Student does not exist")
		PRINT("Press 0 if you want to cancel or press any key to continue")
		GET cancel_input 
		IF cancel_input EQUALS '0' THEN
			BREAK LOOP 
		END IF 
	END WHILE

	PRINT("Please try using your function now")
	OPEN file ENROLLMENTS_FILE IN WRITE MODE AND CLOSE
	EXIT

DEFINE FUNCTION add_student_id(students)
	WHILE True 
		PRINT("Enter student's ID (TPXXXXXX): ")
		GET new_student_id 
		IF null_input(new_student_id) THEN
			CONTINUE
		ELSE IF cancel_command(new_student_id) THEN
			RETURN None
		END IF 
		FOR EACH student IN students
			IF student[0] EQUALS new_student_id THEN
				PRINT("Invalid input, (" + new_student_id + " already exists).")
				BREAK LOOP
			END IF
		ELSE
			OPEN file STUDENTS_FILE IN READ MODE AS students_file
			DEFINE duplicate_found AS False
			FOR EACH student IN students_file
				SET student_id_from_file TO student.split(",")[0].strip()
				IF new_student_id EQUALS student_id_from_file THEN
					PRINT("Invalid input, (" + new_student_id + " already exists).")

					BREAK LOOP

			END FOR
			IF NOT duplicate_found THEN
				RETURN new_student_id
			END IF 
		END FOR

			RETURN new_student_id
	END WHILE

DEFINE FUNCTION save_students(students_list)

DEFINE update_tuition_record(student_id, student_name, student_program, total_tuition, paid_tuition)
	SET updated_data TO False
	DEFINE records AS LIST
	OPEN file TUITION_FILE IN READ MODE AS file
	FOR EACH line IN file 
		SPLIT line BY "," INTO data
		IF data[0] EQUALS student_id AND data[1] EQUALS student_name AND data[2] EQUALS student_program THEN
			SET data[3] TO addition(data[3], total_tuition) 
			SET data[4] TO addition(data[4], paid_tuition)
			SET data[5] TO subtraction(data[3], data[4])
			SET updated_data TO True
		END IF
		APPEND data TO records
	END FOR
	IF NOT updated_data THEN
		APPEND [student_id, student_name, student_program, total_tuition, paid_tuition, subtraction(total_tuition, paid_tuition)] TO records
	END IF 
	OPEN file TUITION_FILE IN WRITE MODE AS file
	FOR EACH record IN records
		WRITE record JOINED BY "," TO file
	END FOR 
	RETURN updated_data
END DEFINE
